try {
  if (promiseIsResolved(PhoneDetailsPromise)) {
    return result;
  } else if (promiseIsRejected) {
    throw reason;
  } else {
    throw promise;
  }
} catch (error) {
  if (error instanceof Promise) {
    promise.then(
      (r) => (result = r),
      (e) => (reason = e),
    );
