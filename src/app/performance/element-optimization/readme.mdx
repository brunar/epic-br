# Element Optimization

```tsx
function Message({ greeting }) {
  console.log('rendering greeting', greeting);

  return <div>{greeting}</div>;
}

function Counter() {
  const [count, setCount] = useState(0);
  const increment = () => setCount((c) => c + 1);

  return (
    <div>
      <button onClick={increment}>The count is {count}</button>
      <Message greeting="Hello" />
    </div>
  );
}
```

```tsx
function Message({ greeting }) {

  return <div>{greeting}</div>;
}

const message = <Message greeting="Hello" />;
// Doing that means jsx1 === jsx2 and it does not re-render because the old element is equal the new element "Hello"

  const [count, setCount] = useState(0);
  const increment = () => setCount((c) => c + 1);
```
