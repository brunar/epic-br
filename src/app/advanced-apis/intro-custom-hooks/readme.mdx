#Intro Custom Hooks

```tsx
// Custom Hook

function useCount() {
  const [count, setCount] = useState(0);
  const increment = () => setCount((c) => c + 1);
  return { count, increment };
}

function Counter() {
  const { count, increment } = useCount();

  return <button onClick={increment}>{count}</button>;
}
```

```tsx
function Counter() {
  const { count, increment } = useCount();

  useEffect(() => {
    const id = setInterval(() => {
      increment();
    }, 1000);

    return () => clearInterval(id);
  }, [increment]);

  return <div>{count}</div>;
}
```

```tsx
// This is a naive way of the implementation of what memoize does, it is just to understand a bit

function memoize<ArgType, ReturnValue>(cb: (arg: ArgType) => ReturnValue) {
  const cache: Record<ArgType, ReturnValue> = {};

  return function memoized(arg: ArgType) {
    if (cache[arg] === undefined) {
      cache[arg] = cb(arg);
    }
    return cache[arg];
  };
}

const addOne = memoize((num: number) => num + 1);
const getDog = memoize((name: string) => new Dog(name));
```

```tsx
// useCallback is a special hook that will memoize the function that was given.
// It stores the function that you give it in the first render, the first time we run.
// Stores it in a special place, and so long as the dependency array never changes.
// And it is going to give you back the same one that you gave the first time.
// So memory is being allocated every time this renders for this function.
// Always give you the previous one if the dependency array doesn't change. []

function useCount() {
  const [count, setCount] = useState(0);
  const increment = useCallback(() => setCount((c) => c + 1), []);
  return { count, increment };
}

// This means the [increment] will be the same every single time

function Counter() {
  const { count, increment } = useCount();

  useEffect(() => {
    const id = setInterval(increment, 1000);

    return () => clearInterval(id);
  }, [increment]);

  return <div>{count}</div>;
}
```

```tsx
// This is NOT how React implements this function. That's the ideia what's going on!
// More like that if the dependency changes [] it will return the new callback otherwise it returns the previous

let lastCallBack;

function useCallback(callback, deps) {
  if (depsChanged(deps)) {
    lastCallBack = callback;
    return callback;
  } else {
    return lastCallBack;
  }
}
```
